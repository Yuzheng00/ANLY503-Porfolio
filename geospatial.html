#Texas

p1=ggplot()+geom_sf(data=us_states, fill=NA)+
  geom_sf(data=county_pres %>% filter(STATEFP=='48',party=='democrat'),aes(fill=vote_pro))+
  facet_wrap(~year, ncol =3)+
  coord_sf(xlim = c(-108, -93), ylim = c(23, 35), expand = F)+
  labs(fill = 'proportion of vote')+
  theme_bw() + 
  theme(panel.background = element_rect(fill = "aliceblue"),
        axis.ticks = element_blank(),
        axis.text = element_blank(),
        panel.grid = element_line(color = "white", size = 0.8))

p2=ggplot()+geom_sf(data=us_states, fill=NA)+
  geom_sf(data=county_unemployment %>% filter(STATEFP=='48'),aes(fill=value))+
  facet_wrap(~year, ncol =3)+
  coord_sf(xlim = c(-108, -93), ylim = c(23, 35), expand = F)+
  labs(x = '', y = '', fill = 'unemployment rate')+
  theme_bw() + 
  theme(panel.background = element_rect(fill = "aliceblue"),
        axis.ticks = element_blank(),
        axis.text = element_blank(),
        panel.grid = element_line(color = "white", size = 0.8))

grid.arrange(p1,p2, nrow=2,ncol=1,top = 'US election data and unemployment rates in Texas')

#python

import numpy as np
import pandas as pd
import re
countypres=pd.read_csv('data/countypres_2000-2016.csv')
unemployment=pd.read_excel('data/Unemployment.xls', skiprows=7)

#select data in 2008,2012,2016
countypres =countypres[countypres['year'].isin([2008,2012,2016])]
countypres = countypres[countypres['party'].isin(['republican','democrat'])]
countypres['FIPS']=countypres['FIPS'].astype(str).apply(lambda x: re.sub('\.0','',x))
#add 0
countypres['FIPS'] = countypres['FIPS'].apply(lambda x: str(x).zfill(5))
countypres['vote_pro']=countypres['candidatevotes']/countypres['totalvotes']


#unemployment rate in 2008,2012, 2016
unemployment.rename(columns={'FIPStxt':'FIPS'},inplace=True)
unemployment['FIPS'] = unemployment['FIPS'].astype(str)
unemployment = unemployment[['FIPS','Stabr','area_name','Unemployment_rate_2008','Unemployment_rate_2012','Unemployment_rate_2016']]
